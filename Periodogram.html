$~~~~~~$$\{x_t\}$$x_0,x_1,...,x_N$$N$(Periodogram)$\{x_t\}$$X(m)$

1.伪代码：¶

Input$x_1,x_2,...x_n$ .
Step1: $$X(m)=FFT\{x_1,x_2,...,x_n\},m=0,1,2..,N$$

Step2: $$P(\frac{m\pi}{N+1})=\frac{\mid X(m)\mid^2}{N}$$

Output:$P(\frac{m\pi}{N+1})$$[0,\pi]$

import numpy as np
import matplotlib.pyplot as plt
from scipy.fftpack import fft
import math
pi=math.pi
#N
def generate_sin(N):
    '''
    :param 
            N: number of sample
    :return: 
            xn:the A series of signals#
    '''
    # sample rate
    fs = 1000
    # N 
    t = np.linspace(0, 0.3, N)
    # 
    A = np.array([2, 8]).reshape(-1, 1)
    # 
    f = np.array([150, 140]).reshape(-1, 1)
    # 
    xn = (A * np.sin(2 * np.pi * f * t)).sum(axis=0) + 5 * np.random.randn(*t.shape)
    return xn
'''
Step1:generate the a random signals
'''
# number of sample
N = 301
# 
x = generate_sin(N)
#
plt.plot(x)
#
plt.title("Original Signal Plot")
#x
plt.xlabel("Frenquency")
plt.ylabel("Amplitude")
plt.show()
'''
Step2:Discrete Fourier Transform to x
'''
#x
X = fft(x)
# Magnitude(amplitude)
mX = np.abs(X)
# phase
pX = np.angle(X)
'''
 Step3:Calculate the |X(m)|^2/N
'''
P = (1/N)*np.square(mX)
'''
Step4: Plot the graph
'''
#
norm_P =[]
#
max_P = np.max(P)
#
min_P = np.min(P)
#P(t)
for t in P:
    t = (t-min_P)/(max_P-min_P)
    norm_P.append(t)
#
plt.plot(norm_P)
#
plt.title("Periodogram")
#x
plt.xlabel("Frenquency")
#y
plt.ylabel("power spectrum")
plt.show()
